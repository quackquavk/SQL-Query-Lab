<task type="auto">
<name>Task 2: Add API client functions for admin features</name>
<files>scripts/apiClient.js</files>
<action>
Add to `scripts/apiClient.js` the following functions:

**SQL Agent Jobs:**
```javascript
// List all jobs with category hierarchy
export async function fetchSqlAgentJobs() {
  const res = await fetch(`/api/sql-agent/jobs/${currentDb}`)
  return res.json()
}

// Get job details (steps, schedules, alerts)
export async function fetchJobDetails(jobName) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}`)
  return res.json()
}

// Get job history (paginated, 50 per page)
export async function fetchJobHistory(jobName, page = 0) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}/history?page=${page}&pageSize=50`)
  return res.json()
}

// Job control actions
export async function startJob(jobName) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}/start`, { method: 'POST' })
  return res.json()
}

export async function stopJob(jobName) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}/stop`, { method: 'POST' })
  return res.json()
}

export async function enableJob(jobName) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}/enable`, { method: 'POST' })
  return res.json()
}

export async function disableJob(jobName) {
  const res = await fetch(`/api/sql-agent/job/${currentDb}/${encodeURIComponent(jobName)}/disable`, { method: 'POST' })
  return res.json()
}
```

**Backup/Restore:**
```javascript
// Execute backup (returns JSON, WebSocket for progress streaming)
export async function executeBackup(options) {
  const res = await fetch('/api/backup', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(options)
  })
  return res.json()
}

// List backup history for a database
export async function fetchBackupHistory(dbName) {
  const res = await fetch(`/api/backup/history/${encodeURIComponent(dbName)}`)
  return res.json()
}

// Execute restore
export async function executeRestore(options) {
  const res = await fetch('/api/restore', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(options)
  })
  return res.json()
}

// Verify backup file integrity
export async function verifyBackup(backupPath) {
  const res = await fetch('/api/restore/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ backupPath })
  })
  return res.json()
}
```

Follow existing API client patterns in the file (base URL handling, JSON responses, error handling via throw on non-OK).
