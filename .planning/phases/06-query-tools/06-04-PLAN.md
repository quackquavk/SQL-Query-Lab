```javascript
import { Hono } from 'hono';

const optimize = new Hono();

// Execute query with SET SHOWPLAN_XML ON
// Parse resulting XML for optimizer warnings and missing indexes
optimize.post('/', async (c) => {
  const { sql, database } = await c.req.json();
  
  try {
    // 1. Execute SET SHOWPLAN_XML ON
    // 2. Run user query
    // 3. Capture XML Showplan
    // 4. Parse for suggestions
    
    const suggestions = analyzeQuery(sql, xmlPlan);
    
    return c.json({ suggestions });
  } catch (err) {
    return c.json({ error: err.message }, 500);
  }
});

export function analyzeQuery(sql, xmlPlan) {
  const suggestions = [];
  const parser = new DOMParser();
  const doc = parser.parseFromString(xmlPlan, 'text/xml');
  
  // Extract missing indexes
  const missingIndexGroups = doc.querySelectorAll('MissingIndexGroup');
  missingIndexGroups.forEach(group => {
    const index = group.querySelector('MissingIndex');
    const columns = index.querySelectorAll('Column');
    const impact = group.getAttribute('Impact') || 'Unknown';
    
    suggestions.push({
      type: 'index',
      message: `Consider adding index on ${[...columns].map(c => c.getAttribute('Name')).join(', ')}`,
      impact: parseFloat(impact),
      createStatement: `CREATE INDEX ix_${index.getAttribute('Name') || 'missing'} ON ${index.getAttribute('Database')}.${index.getAttribute('Schema')}.${index.getAttribute('Table')} ([${[...columns].map(c => c.getAttribute('Name')).join('], [')}]);`,
      line: 1,
      column: 1,
      sql: null
    });
  });
  
  // Extract other warnings (table scans, no statistics, etc.)
  const warnings = doc.querySelectorAll('Warnings, RelOp[EstimateRows="0"]');
  warnings.forEach(w => {
    suggestions.push({
      type: 'restructure',
      message: w.getAttribute('Warning') || 'Consider optimizing this operation',
      line: 1,
      column: 1,
      sql: null
    });
  });
  
  return suggestions;
}

export default optimize;
```

```javascript
// Fetch optimization suggestions for SQL
export async function fetchOptimizationSuggestions(sql, database) {
  const res = await fetch('/api/optimize', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ sql, database })
  });
  const data = await res.json();
  if (data.error) throw new Error(data.error);
  return data;
}

// Fetch and parse missing indexes from execution plan
export async function fetchMissingIndexes(query, database) {
  // First get execution plan XML
  const planRes = await fetch('/api/execution-plan', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ query, database })
  });
  const planData = await planRes.json();
  
  if (!planData.xml) return { missingIndexes: [] };
  
  // Parse missing indexes from XML
  const parser = new DOMParser();
  const doc = parser.parseFromString(planData.xml, 'text/xml');
  
  const missingIndexes = [];
  const groups = doc.querySelectorAll('MissingIndexGroup');
  
  groups.forEach(group => {
    const index = group.querySelector('MissingIndex');
    if (!index) return;
    
    const columns = [...index.querySelectorAll('Column')].map(c => c.getAttribute('Name'));
    const impact = parseFloat(group.getAttribute('Impact') || '0');
    
    missingIndexes.push({
      name: index.getAttribute('Name') || 'missing_index',
      table: index.getAttribute('Table') || '',
      schema: index.getAttribute('Schema') || 'dbo',
      database: index.getAttribute('Database') || database,
      columns,
      impact,
      createStatement: `CREATE INDEX ix_${index.getAttribute('Name') || 'missing'} ON [${index.getAttribute('Database')}].[${index.getAttribute('Schema')}].[${index.getAttribute('Table')}] ([${columns.join('], [')}]);`
    });
  });
  
  return { missingIndexes, xml: planData.xml };
}
```

```javascript
function showMissingIndexTooltip(event, indexInfo) {
```
