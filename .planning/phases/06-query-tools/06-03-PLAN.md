```javascript
let _editor = null;
let _decorations = [];
let _tooltipEl = null;

// enableOptimizationHints(editor, suggestions)
// suggestions: from backend API
// line/column are 1-indexed
export function enableOptimizationHints(editor, suggestions) {
  _editor = editor;
  clearOptimizationDecorations();
  
  suggestions.forEach(s => {
    const line = s.line - 1; // CodeMirror is 0-indexed
    const from = CodeMirror.Pos(line, s.column - 1);
    const to = CodeMirror.Pos(line, s.column - 1 + s.length);
    
    // Create wavy amber underline decoration
    const deco = editor.markText(from, to, {
      className: 'optimization-highlight',
      title: s.message
    });
    
    _decorations.push({ deco, suggestion: s });
  });
  
  // Add click handler for tooltips
  editor.on('click', handleHighlightClick);
}

// Clear all optimization decorations
export function clearOptimizationDecorations() {
  _decorations.forEach(d => d.deco.clear());
  _decorations = [];
  if (_tooltipEl) {
    _tooltipEl.remove();
    _tooltipEl = null;
  }
}

// Handle click on highlighted text
function handleHighlightClick(cm, e) {
  // Find if clicked position is within a decoration
  // If yes: show tooltip with suggestion details
}

// Show tooltip with suggestion text
function showTooltip(event, suggestion) {
  if (!_tooltipEl) {
    _tooltipEl = document.createElement('div');
    _tooltipEl.className = 'optimization-tooltip';
    document.body.appendChild(_tooltipEl);
  }
  
  _tooltipEl.innerHTML = `
    <div class="optimization-tooltip-type">${suggestion.type.toUpperCase()}</div>
    <div class="optimization-tooltip-message">${suggestion.message}</div>
    ${suggestion.sql ? `<div class="optimization-tooltip-sql">${suggestion.sql}</div>` : ''}
  `;
  
  _tooltipEl.style.display = 'block';
  _tooltipEl.style.left = (event.pageX + 10) + 'px';
  _tooltipEl.style.top = (event.pageY + 10) + 'px';
}

function hideTooltip() {
  if (_tooltipEl) {
    _tooltipEl.style.display = 'none';
  }
}
```

async function onQueryComplete(results) {
  // Existing: render results
  // New: fetch optimization suggestions for this SQL
  if (runtime.cursor.currentMode === 'mssql') {
    const sql = editor.getValue();
    const db = runtime.cursor.currentDbName;
    try {
      const { suggestions } = await fetchOptimizationSuggestions(sql, db);
      if (suggestions && suggestions.length > 0) {
        enableOptimizationHints(editor, suggestions);
      } else {
        clearOptimizationDecorations();
      }
    } catch (e) {
      console.warn('Optimization fetch failed:', e);
    }
  }
}

function showTooltip(event, suggestion) {
  // Position with offset to avoid cursor overlap
  let left = event.pageX + 15;
  let top = event.pageY + 15;
  
  // If tooltip would go off right edge, flip to left of cursor
  if (left + 300 > window.innerWidth) {
    left = event.pageX - 320;
  }
  
  // If tooltip would go off bottom, flip above
  if (top + 200 > window.innerHeight) {
    top = event.pageY - 220;
  }
  
  _tooltipEl.style.left = left + 'px';
  _tooltipEl.style.top = top + 'px';
  
  // Add close on click outside
  document.addEventListener('click', closeTooltipOnClickOutside);
}

function closeTooltipOnClickOutside(e) {
  if (!_tooltipEl.contains(e.target) && !e.target.classList.contains('optimization-highlight')) {
    hideTooltip();
    document.removeEventListener('click', closeTooltipOnClickOutside);
  }
}
