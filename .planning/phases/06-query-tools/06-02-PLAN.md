**Bar Chart:**
```javascript
export function renderBarChart(data, xCol, yCol) {
  const d3 = window.d3;
  const container = document.getElementById('chartContainer');
  const svg = d3.select(container).append('svg')
    .attr('width', '100%')
    .attr('height', 200);

  // Extract data
  const xValues = data.rows.map(r => r[xCol]);
  const yValues = data.rows.map(r => parseFloat(r[yCol]));
  
  // Scales
  const x = d3.scaleBand().domain(xValues).range([40, 300]).padding(0.2);
  const y = d3.scaleLinear().domain([0, d3.max(yValues)]).range([180, 10]);
  
  // Bars
  svg.selectAll('.chart-bar')
    .data(data.rows)
    .enter().append('rect')
    .attr('class', 'chart-bar')
    .attr('x', d => x(d[xCol]))
    .attr('y', d => y(parseFloat(d[yCol])))
    .attr('width', x.bandwidth())
    .attr('height', d => 180 - y(parseFloat(d[yCol])))
    .attr('fill', '#e8a030');
  
  // Axes
  svg.append('g').attr('transform', 'translate(0,180)').call(d3.axisBottom(x));
  svg.append('g').attr('transform', 'translate(40,0)').call(d3.axisLeft(y));
}
```

**Line Chart:** Same structure but with .line path and .area fill below line.

**Pie Chart:**
```javascript
export function renderPieChart(data, xCol, yCol) {
  const d3 = window.d3;
  const radius = 80;
  const pie = d3.pie().value(d => parseFloat(d[yCol]));
  const arc = d3.arc().innerRadius(0).outerRadius(radius);
  
  const svg = d3.select('#chartContainer').append('svg')
    .attr('width', 400).attr('height', 200);
  
  const g = svg.append('g').attr('transform', 'translate(100,100)');
  
  g.selectAll('.chart-slice')
    .data(pie(data.rows))
    .enter().append('path')
    .attr('class', 'chart-slice')
    .attr('d', arc)
    .attr('fill', '#e8a030');
}
```
