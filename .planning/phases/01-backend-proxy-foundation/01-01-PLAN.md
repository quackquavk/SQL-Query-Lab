**Connection Pool Pattern (CRITICAL — Anti-Pattern 3 from research):**
```javascript
import mssql from 'mssql';

// Per-user-per-server pool registry
// Key: `${userId}:${server}` → Pool instance
const pools = new Map();

async function getPool(userId, server, authType, credentials) {
  const key = `${userId}:${server}`;
  
  if (pools.has(key)) {
    return pools.get(key);
  }
  
  const config = buildConfig(server, authType, credentials);
  const pool = new mssql.ConnectionPool(config);
  await pool.connect();
  
  // Cap per-user concurrent connections (per phase context: max 5)
  pool.on('error', (err) => {
    console.error('Pool error:', err.message);
    pools.delete(key);
  });
  
  pools.set(key, pool);
  return pool;
}

function buildConfig(server, authType, credentials) {
  const config = {
    server: server,
    options: {
      encrypt: true,        // Azure SQL requires this
      trustServerCertificate: false,
      port: 1433
    },
    pool: {
      max: 5,              // Per user per server
      min: 0,
      idleTimeoutMillis: 30000
    }
  };
  
  switch (authType) {
    case 'sql':
      config.user = credentials.username;
      config.password = credentials.password;
      config.database = credentials.database || 'master';
      break;
    case 'windows':
      // NTLM/Kerberos via SPNEGO — backend handles
      config.authentication = {
        type: 'ntlm',
        options: {
          domain: credentials.domain || ''
        }
      };
      config.database = credentials.database || 'master';
      break;
    case 'entra':
      // Entra ID — use access token from credentials
      config.authentication = {
        type: 'azure-active-directory-service-principal',
        options: {
          tenantId: credentials.tenantId,
          clientId: credentials.clientId,
          clientSecret: credentials.clientSecret
        }
      };
      config.database = credentials.database || 'master';
      break;
  }
  
  return config;
}

async function testConnection(connConfig) {
  const pool = new mssql.ConnectionPool(buildConfig(...connConfig));
  await pool.connect();
  const result = await pool.query('SELECT @@VERSION as version');
  pool.close();
  return { serverVersion: result.recordset[0].version };
}

async function executeQuery(pool, sql, params = []) {
  // VALIDATION: Parameterize all queries (PROF-02, D-13)
  // Never execute raw string interpolation — use mssql parameterized queries
  
  const request = pool.request();
  
  // Add parameters safely
  for (const param of params) {
    request.input(param.name, param.type, param.value);
  }
  
  return request.query(sql);
}

export { getPool, buildConfig, testConnection, executeQuery };
```

Create backend/services/crypto.js — User-derived key encryption/decryption for connection credentials.

**Encryption approach (per D-04/D-05):**
```javascript
import { randomBytes, createCipheriv, createDecipheriv, pbkdf2Sync } from 'crypto';

const ALGORITHM = 'aes-256-gcm';
const KEY_LENGTH = 32;
const SALT_LENGTH = 32;
const IV_LENGTH = 16;
const AUTH_TAG_LENGTH = 16;
const PBKDF2_ITERATIONS = 100000; // Per D-14 note (>100k)

/**
 * Derive encryption key from user's master password.
 * Uses PBKDF2 with high iteration count (per phase context).
 * Key is NEVER stored — derived on each unlock.
 */
function deriveKey(masterPassword, salt) {
  return pbkdf2Sync(
    masterPassword,
    salt,
    PBKDF2_ITERATIONS,
    KEY_LENGTH,
    'sha256'
  );
}

/**
 * Encrypt connection credentials blob.
 * Returns: { salt, iv, authTag, encrypted } — all base64 encoded.
 * 
 * Per D-04: user-derived key, never stored.
 * Per D-05: stored as encrypted blob server-side.
 */
async function encryptConnection(connectionData, masterPassword) {
  const salt = randomBytes(SALT_LENGTH);
  const iv = randomBytes(IV_LENGTH);
  const key = deriveKey(masterPassword, salt);
  
  const cipher = createCipheriv(ALGORITHM, key, iv);
  
  const encrypted = Buffer.concat([
    cipher.update(JSON.stringify(connectionData), 'utf8'),
    cipher.final()
  ]);
  
  const authTag = cipher.getAuthTag();
  
  return {
    salt: salt.toString('base64'),
    iv: iv.toString('base64'),
    authTag: authTag.toString('base64'),
    encrypted: encrypted.toString('base64')
  };
}

/**
 * Decrypt connection credentials blob.
 * Requires the same masterPassword used for encryption.
 */
async function decryptConnection(blob, masterPassword) {
  const salt = Buffer.from(blob.salt, 'base64');
  const iv = Buffer.from(blob.iv, 'base64');
  const authTag = Buffer.from(blob.authTag, 'base64');
  const encrypted = Buffer.from(blob.encrypted, 'base64');
  
  const key = deriveKey(masterPassword, salt);
  
  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(authTag);
  
  const decrypted = Buffer.concat([
    decipher.update(encrypted),
    decipher.final()
  ]);
  
  return JSON.parse(decrypted.toString('utf8'));
}
```

```javascript
export function handleQueryWebSocket(ws, ctx) {
  // Parse incoming message
  const { type, connectionId, sql, params, queryId } = msg;
  
  if (type !== 'execute') return;
  
  // Validate session (user must be authenticated)
  const userId = getSessionUserId(ctx);
  if (!userId) {
    ws.send(JSON.stringify({ type: 'error', message: 'Unauthorized', queryId }));
    return;
  }
  
  // Load decrypted connection credentials
  const connection = await getConnection(connectionId, userId);
  if (!connection) {
    ws.send(JSON.stringify({ type: 'error', message: 'Connection not found', queryId }));
    return;
  }
  
  // Decrypt credentials with user's master password
  const credentials = await decryptConnection(connection.blob, userMasterPassword);
  
  // Get or create connection pool
  const pool = await getPool(userId, connection.server, connection.authType, credentials);
  
  // Query cancellation support
  let cancelled = false;
  ws.on('message', (msg) => {
    const cmd = JSON.parse(msg);
    if (cmd.type === 'cancel' && cmd.queryId === queryId) {
      cancelled = true;
    }
  });
  
  try {
    // Execute parameterized query (PROF-02 — validates all queries)
    const result = await executeQuery(pool, sql, params);
    
    if (cancelled) {
      ws.send(JSON.stringify({ type: 'error', message: 'Query cancelled', queryId }));
      return;
    }
    
    // Stream columns first
    const columns = result.columns.map(col => ({
      name: col.name,
      type: getTsqlTypeName(col.type)
    }));
    ws.send(JSON.stringify({ type: 'columns', columns, queryId }));
    
    // Stream rows in batches
    const BATCH_SIZE = 100;
    for (let i = 0; i < result.recordset.length; i += BATCH_SIZE) {
      if (cancelled) break;
      const batch = result.recordset.slice(i, i + BATCH_SIZE);
      ws.send(JSON.stringify({
        type: 'rows',
        rows: batch,
        total: result.recordset.length,
        queryId
      }));
    }
    
    ws.send(JSON.stringify({
      type: 'done',
      rowsAffected: result.rowsAffected?.[0] || 0,
      executionTime: result.executionTime || 0,
      queryId
    }));
    
  } catch (err) {
    // Decoded error messages (not raw TDS codes) — EXEC-05
    ws.send(JSON.stringify({
      type: 'error',
      message: decodeSqlError(err),
      code: err.code,
      queryId
    }));
  }
}
```

import { Client } from '@microsoft/microsoft-authentication-library-for-node';

const EntraAuth = {
  client: null,
  
  /**
   * Initiate device code flow for Entra ID authentication.
   * Returns { deviceCode, userCode, verificationUrl, expiresAt }
   */
  async initiateDeviceFlow(config) {
    const client = new Client({
      auth: {
        clientId: config.clientId,
        authority: `https://login.microsoftonline.com/${config.tenantId}`
      }
    });
    
    const result = await client.acquireTokenByDeviceCode({
      scopes: ['https://database.windows.net/.default']
    });
    
    return {
      accessToken: result.accessToken,
      refreshToken: result.refreshToken,
      expiresAt: result.expiresOn
    };
  },
  
  /**
   * Refresh access token using refresh token.
   */
  async refreshToken(config, refreshToken) {
    const client = new Client({ /* same config */ });
    const result = await client.acquireTokenByRefreshToken({
      scopes: ['https://database.windows.net/.default'],
      refreshToken
    });
    return {
      accessToken: result.accessToken,
      expiresAt: result.expiresOn
    };
  },
  
  /**
   * Get SQL Server access token for mssql connection.
   * Pass to mssql as accessToken in authentication options.
   */
  async getAccessToken(config) {
    if (!this.client) {
      this.client = await this.initiateDeviceFlow(config);
    }
    
    // Check if token expired and refresh if needed
    if (Date.now() > this.client.expiresAt) {
      this.client = await this.refreshToken(config, this.client.refreshToken);
    }
    
    return this.client.accessToken;
  }
};

export default EntraAuth;

/**
 * Session validation middleware.
 * Extracts user session from cookie or Authorization header.
 * Creates ctx.user for downstream handlers.
 */
export async function sessionMiddleware(ctx, next) {
  const sessionId = getCookie(ctx, 'session') || ctx.req.header('Authorization')?.replace('Bearer ', '');
  
  if (!sessionId) {
    ctx.status = 401;
    return ctx.json({ error: 'Unauthorized' });
  }
  
  const user = await validateSession(sessionId);
  if (!user) {
