```javascript
/**
 * API Client for backend proxy communication.
 * fetch for REST, WebSocket for query streaming.

const API_BASE = '/api';
let ws = null;
let wsCallbacks = {};
let queryIdCounter = 0;

// ============================================
// REST API (Connection CRUD)
// ============================================

/**
 * Test connection before saving (CONN-07)
 * @returns {{ success: boolean, serverVersion?: string, error?: string }}
 */
export async function testConnection({ server, database, authType, credentials }) {
  const res = await fetch(`${API_BASE}/connections/test`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    credentials: 'include',
    body: JSON.stringify({ server, database, authType, credentials })
  });
  return res.json();
}

/**
 * Save a new connection (CONN-05)
 * @returns {{ id: string, name: string }}
 */
export async function saveConnection({ name, server, database, authType, credentials }) {
  const res = await fetch(`${API_BASE}/connections`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    credentials: 'include',
    body: JSON.stringify({ name, server, database, authType, credentials })
  });
  return res.json();
}

/**
 * List saved connections (credentials NOT included — PROF-01)
 * @returns {Array<{ id, name, server, authType }>}
 */
export async function listConnections() {
  const res = await fetch(`${API_BASE}/connections`, {
    credentials: 'include'
  });
  return res.json();
}

/**
 * Delete a saved connection (CONN-08)
 */
export async function deleteConnection(id) {
  const res = await fetch(`${API_BASE}/connections/${id}`, {
    method: 'DELETE',
    credentials: 'include'
  });
  return res.json();
}

// ============================================
// WebSocket Query Execution
// ============================================

/**
 * Connect to WebSocket for query execution.
 * Returns promise that resolves when connected.
 */
export function connectQuerySocket({ onColumns, onRows, onDone, onError, onCancel }) {
  return new Promise((resolve, reject) => {
    const protocol = location.protocol === 'https:' ? 'wss:' : 'ws:';
    ws = new WebSocket(`${protocol}//${location.host}/api/query`);
    
    ws.onopen = () => resolve();
    ws.onerror = (e) => reject(new Error('WebSocket connection failed'));
    
    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);
      const { type, queryId } = msg;
      
      switch (type) {
        case 'columns':
          onColumns?.(msg.columns);
          break;
        case 'rows':
          onRows?.(msg.rows, msg.total);
          break;
        case 'done':
          onDone?.(msg.rowsAffected, msg.executionTime);
          break;
        case 'error':
          onError?.(msg.message, msg.code, queryId);
          break;
      }
    };
  });
}

/**
 * Execute query via WebSocket (CONN-01, CONN-02, CONN-03, CONN-04)
 * Returns queryId for cancellation.
 */
export function executeQuery(connectionId, sql, params = []) {
  const queryId = `q-${++queryIdCounter}`;
  
  ws.send(JSON.stringify({
    type: 'execute',
    connectionId,
    sql,
    params,
    queryId
  }));
  
  return queryId;
}

/**
 * Cancel running query.
 */
export function cancelQuery(queryId) {
  ws.send(JSON.stringify({ type: 'cancel', queryId }));
}

/**
 * Close WebSocket connection.
 */
export function disconnectQuerySocket() {
  if (ws) {
    ws.close();
    ws = null;
  }
}
```

```javascript
/**
 * Run query against live SQL Server via backend proxy.
 * Uses WebSocket for streaming results.
 * 
 * @param {string} sql - SQL query to execute
 * @returns {Promise<{ columns, rows, rowsAffected, executionTime }>}
 */
export async function runLiveQuery(sql) {
  const connectionId = runtime.cursor.connectionId;
  if (!connectionId) {
    throw new Error('Not connected to a server');
  }
  
  // Connect WebSocket
  let results = null;
  let error = null;
  
  await connectQuerySocket({
    onColumns: (columns) => {
      results = { columns, rows: [], rowsAffected: 0, executionTime: 0 };
    },
    onRows: (rows, total) => {
      if (results) results.rows.push(...rows);
    },
    onDone: (rowsAffected, executionTime) => {
      if (results) {
        results.rowsAffected = rowsAffected;
        results.executionTime = executionTime;
      }
    },
    onError: (message, code, queryId) => {
      error = { message, code };
    }
  });
  
  // Execute query
  const queryId = executeQuery(connectionId, sql, []);
  
  // Wait for completion (promise resolves when 'done' received)
  await new Promise((resolve, reject) => {
    const timeout = setTimeout(() => {
      cancelQuery(queryId);
      reject(new Error('Query timeout (30s)'));
    }, 30000);
    
    // Poll for results completion
    const check = () => {
      if (error) {
        clearTimeout(timeout);
        disconnectQuerySocket();
        reject(new Error(error.message));
        return;
      }
      if (results && results.executionTime > 0) {
        clearTimeout(timeout);
        disconnectQuerySocket();
        resolve(results);
        return;
      }
      setTimeout(check, 50);
    };
    check();
  });
  
  return results;
}
```
