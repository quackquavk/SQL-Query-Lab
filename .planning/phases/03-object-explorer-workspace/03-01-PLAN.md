
    **apiClient Extensions (scripts/apiClient.js):**
    Add to apiClient:
    ```javascript
    export async function fetchObjectTree(connectionId) {
      const res = await fetch(`/api/schema?connectionId=${connectionId}`);
      return res.json();  // Returns { databases: [{ name, tables, views, procedures, functions }] }
    }

    export async function fetchTableColumns(connectionId, database, table) {
      const res = await fetch(`/api/schema/${database}/${table}/columns?connectionId=${connectionId}`);
      return res.json();  // Returns { columns: [{ name, dataType, isNullable, isPrimaryKey }] }
    }

    export async function fetchProcedureDefinition(connectionId, database, procedure) {
      const res = await fetch(`/api/schema/${database}/${procedure}/definition?connectionId=${connectionId}`);
      return res.text();  // Returns SQL definition string
    }
    ```

    **Tree Render Function (ui.js):**

    **Object Explorer Init (main.js):**

    ```javascript
    }
    ```
