```javascript
// D3 + dagre initialization pattern
export function initErDiagram(svgSelector, schema) {
  // Create dagre graph
  const g = new dagre.graphlib.Graph();
  g.setGraph({ rankdir: 'TB', nodesep: 40, ranksep: 60 });
  g.setDefaultEdgeLabel(() => ({}));

  // Add nodes (tables)
  schema.tables.forEach(t => g.setNode(t.name, { width: 180, height: /* computed */ }));

  // Add edges (FK relationships)
  schema.relationships.forEach(r => g.setEdge(r.from.table, r.to.table));

  // Compute layout
  dagre.layout(g);

  // Render with D3 SVG
  const svg = d3.select(svgSelector);
  // ... render nodes and edges
}

// D3 zoom behavior
export function setupZoom(svgSelector) {
  d3.select(svgSelector)
    .call(d3.zoom()
      .scaleExtent([0.25, 4])
      .on('zoom', (event) => { /* handle pan/zoom */ }));
}
```

From scripts/apiClient.js (extension target):
```javascript
export async function fetchSchema(database) {
  const res = await fetch(`/api/schema/${database}`);
  return res.json();
}

export async function executeDdl(ddl) {
  const res = await fetch('/api/execute-ddl', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ ddl })
  });
  return res.json();
}
```
