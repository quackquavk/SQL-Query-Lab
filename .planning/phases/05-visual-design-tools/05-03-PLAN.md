// GO batch splitting
export function splitBatches(spText) {
  // Split on \nGO\n or \nGO\n; or \ngo\n etc (case insensitive)
  const batchSeparator = /\nGO\s*;?\n/gi;
  const parts = spText.split(batchSeparator);
  return parts.filter(p => p.trim().length > 0);
}

// Parameter extraction from CREATE PROCEDURE signature
export function extractParameters(spText) {
  const match = spText.match(/CREATE\s+PROCEDURE\s+(\w+)\s+([\s\S]*?)\s+AS\s+/i);
  if (!match) return [];
  const paramsStr = match[2];
  const paramMatches = paramsStr.matchAll(/@(\w+)\s+(\w+(?:\(\d+\))?)\s*(=\s*([^,\n]+))?/gi);
  return [...paramMatches].map(m => ({
    name: m[1],
    type: m[2],
    default: m[4]?.trim()
  }));
}

// T-SQL validation via backend
export async function validateTsql(spText) {
  const res = await fetch('/api/validate-tsql', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ sql: spText })
  });
  return res.json(); // { valid: true } or { valid: false, errors: [...] }
}
